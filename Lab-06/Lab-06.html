Segment imagen using k-means clustering¶

We will employ sklearn Kmeans function to run the k-means clustering algorithm. After that, we will take the centroids color and use them to segment the image by replacing each pixel with the closest centroid color.

import cv2
import numpy as np
import matplotlib.pyplot as plt
from sklearn.cluster import KMeans

def v3_distance(v1, v2):
    return np.sqrt(np.sum((v1 - v2)**2))
def filter_colors_k_means(n_clusters, img):
    clt = KMeans(n_clusters=n_clusters)
    clt.fit(img.reshape(-1,3))
    colors_to_map = clt.cluster_centers_
    
    for i in range(img.shape[0]):
        for j in range(img.shape[1]):
            pixel = img[i,j]
            lowest_distance = 999
            nearest_color = None
            for color in colors_to_map:
                distance = v3_distance(color, pixel)
                if distance < lowest_distance:
                    lowest_distance = distance
                    nearest_color = color
            img[i,j] = nearest_color    
    return img

# convert to RGB
img_1 = cv2.cvtColor(img_1, cv2.COLOR_BGR2RGB)
img_2 = cv2.cvtColor(img_2, cv2.COLOR_BGR2RGB)

# filter colors
img_1 = filter_colors_k_means(2, img_1)
img_2 = filter_colors_k_means(3, img_2)

from sklearn.cluster import MeanShift, estimate_bandwidth


def filter_colors_mean_shift(q, img):
    flat_image = img.reshape((-1,3))
    flat_image = np.float32(flat_image)

    bandwidth = estimate_bandwidth(flat_image, quantile=q, n_samples=100)
    ms = MeanShift(bandwidth=bandwidth, max_iter=10, bin_seeding=True)
    ms.fit(flat_image)
    labeled=ms.labels_
    
    colors_to_map = ms.cluster_centers_
    
    for i in range(img.shape[0]):
        for j in range(img.shape[1]):
            pixel = img[i,j]
            lowest_distance = 999
            nearest_color = None

                distance = v3_distance(color, pixel)
                if distance < lowest_distance:
                    lowest_distance = distance
                    nearest_color = color
            img[i,j] = nearest_color    
    return img

Lungs segmentation using Region Growing¶

We will employ a hand-crafted function of region growth to segment the lungs from the thresholded X-ray lungs image from practicum 05. The function will take the seed point,the color and a threshold as input and will return the segmented image.

The first approach for this exercise will come with hand-picked seeds.

def region_growing(img, seed, threshold, color):
    img = img.copy()
    segmented = np.zeros(img.shape)
    black = np.array([0, 0, 0], dtype=np.float64)
    segmented[seed[0], seed[1]] = color
    current_region = [seed]
    while current_region:
        new_region = []
        for pixel in current_region:
            # fill region using neighbors
            for i in range(-1, 2):
                for j in range(-1, 2):
                    if pixel[0]+i >= 0 and pixel[0]+i < img.shape[0] and pixel[1]+j >= 0 and pixel[1]+j < img.shape[1]:
                        # if the pixel is not segmented and the difference between the pixel and the seed is less than the threshold
                        is_segmented = (segmented[pixel[0]+i, pixel[1]+j] == black).all()
                        is_similar = (v3_distance(img[pixel[0], pixel[1]], img[pixel[0]+i, pixel[1]+j]) < threshold).all()

                        if (is_segmented and is_similar):
                            segmented[pixel[0]+i, pixel[1]+j] = color
                            new_region.append([pixel[0]+i, pixel[1]+j])

        current_region = new_region
    return segmented

def multi_seed_region_growing(img, seeds, threshold, colors):
    if len(seeds)!=len(colors):
        print("Error: seeds and colors must have the same length")
        return
    segmented = np.zeros(img.shape, dtype=np.float64)
    for i in range(len(seeds)):
        segmented += region_growing(img, seeds[i], threshold, colors[i])
    return segmented

import random
# automatic seeding process
# we will set a seed everytime the intensity difference of the pixel is greater than 100
# we will parse the image from left to right by its middle row by ten pixels each step
img = cv2.imread('lungs.png', cv2.IMREAD_COLOR)
seeds = []
colors = []
image_center_row = img.shape[0]//2

step = 5
# set image bkg as white
for i in range(step, img.shape[1]-step, step):
    px_i   = img[image_center_row, i]
    px_i_5 = img[image_center_row, i-step]

    if (v3_distance(px_i, px_i_5) > 1):
        seeds.append([image_center_row, i])
        colors.append((random.randrange(0, 255)/255, random.randrange(0, 255)/255, random.randrange(0, 255)/255))
